/**

  • Definition for a binary tree node.
  • struct TreeNode {
  • int val;
  • TreeNode *left;
  • TreeNode *right;
  • TreeNode() : val(0), left(nullptr), right(nullptr) {}
  • TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
  • TreeNode(int x, TreeNode left, TreeNode right) : val(x), left(left), right(right) {}
  • };
    */

    define INT_MAX 2147483647

    define INT_MIN -INT_MAX - 1

    class Solution {
    private:
    bool isValidBST(TreeNode* root, int upper_bound, int lower_bound, bool has_ub, bool has_lb){
     if (root == nullptr) return true;
 else if ((has_ub && root->val >= upper_bound) || (has_lb && root->val <= lower_bound)) return false;
 else return isValidBST(root->left, root->val, lower_bound, true, has_lb)
     & isValidBST(root->right, upper_bound, root->val, has_ub, true);
    }

public:
bool isValidBST(TreeNode* root) {
if (root == nullptr) return true;
else return isValidBST(root->left, root->val, INT_MIN, true, false)
& isValidBST(root->right, INT_MAX, root->val, false, true);
}
};

/**

  • Definition for a binary tree node.
  • struct TreeNode {
  • int val;
  • TreeNode *left;
  • TreeNode *right;
  • TreeNode() : val(0), left(nullptr), right(nullptr) {}
  • TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
  • TreeNode(int x, TreeNode left, TreeNode right) : val(x), left(left), right(right) {}
  • };
    */

    define INT_MAX 2147483647

    define INT_MIN -INT_MAX - 1

    struct queElem{
    TreeNode root;
    int lowerBound;
    int upperBound;
    bool hasLB;
    bool hasUB;
    };
    class Solution {
    private:
    queue que;
    void update(TreeNode     root, int lowerBound, int upperBound, bool hasLB, bool hasUB){
     que.push((queElem){root, lowerBound,upperBound, hasLB, hasUB});
    }

public:
bool isValidBST(TreeNode* root) {
update(root, INT_MIN, INT_MAX, false, false);
while (!que.empty()){
queElem head = que.front();
que.pop();
if (head.root == nullptr) continue;
if ((head.hasUB && head.root->val >= head.upperBound) || (head.hasLB && head.root->val <= head.lowerBound)) return false;
else{
update(head.root->right, head.root->val, head.upperBound, true, head.hasUB);
update(head.root->left, head.lowerBound, head.root->val, head.hasLB, true);
}
}
return true;
}
};

/**

  • Definition for a binary tree node.
  • struct TreeNode {
  • int val;
  • TreeNode *left;
  • TreeNode *right;
  • TreeNode() : val(0), left(nullptr), right(nullptr) {}
  • TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
  • TreeNode(int x, TreeNode left, TreeNode right) : val(x), left(left), right(right) {}
  • };
    /
    class Solution {
    private:
    bool hasPrev;
    int prev;
    bool inorder(TreeNode     root){
     if (root == nullptr) return true;
 else if (!inorder(root->left)) return false;
 else if (hasPrev && prev >= root->val) return false;
 else{
     hasPrev = true;
     prev = root->val;
     return inorder(root->right);
 }
    }
    public:
    bool isValidBST(TreeNode* root) {
     hasPrev = false;
 return inorder(root);
    }
    };
