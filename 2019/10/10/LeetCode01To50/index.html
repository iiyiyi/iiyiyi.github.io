sublen = [1 for i in range(len(s))]

    # The length of the longest sbustring with s[i] as the last character
    if len(s)>0:
        ans = 1
    else:
        ans = 0
    for i in range(1, len(s)):
        sublen[i] = sublen[i-1] + 1
        for j in range(1, sublen[i]+1):
            if (s[i-j] == s[i]):
                sublen[i] = j
                break
        ans = max(ans, sublen[i])
    return ans
